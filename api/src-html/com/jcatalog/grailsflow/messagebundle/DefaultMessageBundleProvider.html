001package com.jcatalog.grailsflow.messagebundle;
002
003import org.springframework.context.i18n.LocaleContextHolder;
004
005import java.util.Locale;
006
007import com.jcatalog.grailsflow.messagebundle.i18n.I18nMessageBundleProvider;
008import com.jcatalog.grailsflow.messagebundle.i18n.I18nMessageBundle;
009
010
011/**
012 * It is default implementation of {@link MessageBundleProvider}. It uses {@link
013 * com.jcatalog.grailsflow.messagebundle.i18n.I18nMessageBundleProvider} for loading message bundle by its name. Also it uses
014 * Spring <code>LocaleContextHolder</code> for getting current Locale - it is used for
015 * creating instance of {@link MessageBundle}.
016 *
017 * @author Roman Denisyuk
018 *
019 * @see com.jcatalog.grailsflow.messagebundle.i18n.I18nMessageBundleProvider
020 * @see DefaultMessageBundle
021 */
022public class DefaultMessageBundleProvider implements MessageBundleProvider {
023    private I18nMessageBundleProvider i18nMessageBundleProvider;
024
025    public MessageBundle getMessageBundle(String bundleName, Locale locale) {
026        // load bundle
027        I18nMessageBundle i18nMessageBundle = i18nMessageBundleProvider
028                .getI18nMessageBundle(bundleName);
029
030        // return bundle using specified locale
031        return new DefaultMessageBundle(i18nMessageBundle, locale);
032    }
033
034    public void setI18nMessageBundleProvider(
035        I18nMessageBundleProvider i18nMessageBundleProvider) {
036        this.i18nMessageBundleProvider = i18nMessageBundleProvider;
037    }
038}
