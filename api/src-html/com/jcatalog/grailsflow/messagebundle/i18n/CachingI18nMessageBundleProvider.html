001package com.jcatalog.grailsflow.messagebundle.i18n;
002
003import org.apache.commons.logging.Log;
004import org.apache.commons.logging.LogFactory;
005
006import java.util.HashMap;
007import java.util.Map;
008
009/**
010 * Abstract implementation {@link I18nMessageBundleProvider} - it offers simple cache for message budles.
011 * 
012 * @author Roman Denisyuk
013 */
014public abstract class CachingI18nMessageBundleProvider implements I18nMessageBundleProvider {
015    private Log log = LogFactory.getLog(CachingI18nMessageBundleProvider.class);
016    private final Map<String, I18nMessageBundle> messageBundles = new HashMap<String, I18nMessageBundle>();
017
018    public I18nMessageBundle getI18nMessageBundle(String bundleName) {
019        I18nMessageBundle messageBundle = null;
020        synchronized (messageBundles) {
021          try{
022            messageBundle = messageBundles.get(bundleName);
023            if (messageBundle == null) {
024                // load message bundle
025                messageBundle = loadMessageBundle(bundleName);
026                // store in cache
027                messageBundles.put(bundleName, messageBundle);
028            }
029          }catch (Throwable ex){
030            log.error("Unexpected exception occurred in synchronized block! Please, contact to administrator. ", ex);
031          }
032        }
033        return messageBundle;
034    }
035
036    /**
037     * Real implementation should implement method of a {@link I18nMessageBundle} loading.
038     * 
039     * @param bundleName name of a message bundle
040     * 
041     * @return loaded I18nMessageBundle
042     */
043    abstract protected I18nMessageBundle loadMessageBundle(String bundleName);
044}
