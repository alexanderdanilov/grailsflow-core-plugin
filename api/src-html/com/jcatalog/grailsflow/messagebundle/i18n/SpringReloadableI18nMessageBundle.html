001package com.jcatalog.grailsflow.messagebundle.i18n;
002
003import org.apache.commons.lang.ArrayUtils;
004
005import org.springframework.context.MessageSource;
006import org.springframework.context.NoSuchMessageException;
007import org.springframework.context.support.ReloadableResourceBundleMessageSource;
008
009import java.util.Locale;
010
011
012/**
013 * Retireves messages from {@link ReloadableResourceBundleMessageSource} that is
014 * passed via constructor.
015 *
016 * @author Roman Denisyuk
017 */
018public class SpringReloadableI18nMessageBundle implements I18nMessageBundle {
019    private MessageSource messageSource;
020
021    public SpringReloadableI18nMessageBundle(
022        ReloadableResourceBundleMessageSource messageSource) {
023        this.messageSource = messageSource;
024    }
025
026    public String getMessage(String key, Locale locale) {
027        return getMessage(key, new String[0], locale);
028    }
029
030    /**
031     * Resolve the given code and arguments as message.
032     *
033     * @param key the key to lookup up, for example 'label.productId'
034     * @param parameters list of parameterss for a message
035     * @param locale Locale
036     *
037     * @return the resolved message, or message code if not found
038     */
039    public String getMessage(String key, String[] parameters, Locale locale) {
040        String message;
041        try {
042            message = messageSource.getMessage(key,
043                    (parameters != null) ? parameters : ArrayUtils.EMPTY_STRING_ARRAY,
044                    locale);
045        } catch (NoSuchMessageException e) {
046            message = "???" + key + "???";
047        }
048        return message;
049    }
050}
