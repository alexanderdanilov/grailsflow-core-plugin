001package com.jcatalog.grailsflow.messagebundle.i18n;
002
003/**
004 * Privides {@link I18nMessageBundle} by name. Name of message bundle is like following: 'com.jcatalog.bundleName'.
005 * 
006 * @author Roman Denisyuk
007 */
008public interface I18nMessageBundleProvider {
009
010    /**
011     * Returns message {@link I18nMessageBundle} implementation by it name and for
012     * specified locale.
013     *
014     * @param bundleName name of a message bundle for example: 'com.jcatalog.bundleName'
015     *
016     * @return {@link I18nMessageBundle}. Message bundle with specified name should be exist.
017     */
018    I18nMessageBundle getI18nMessageBundle(String bundleName);
019}
