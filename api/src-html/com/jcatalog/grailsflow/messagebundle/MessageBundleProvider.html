001package com.jcatalog.grailsflow.messagebundle;
002
003import java.util.Locale;
004
005/**
006 * Privides {@link MessageBundle} by name. Name of message bundle is like following: 'com.jcatalog.bundleName'. 
007 *
008 * @author Roman Denisyuk
009 */
010public interface MessageBundleProvider {
011    /**
012     * Returns message {@link MessageBundle} implementation by it name and for
013     * specified locale.
014     *
015     * @param bundleName name of a message bundle for example: 'com.jcatalog.bundleName'
016     * @param locale should be used for getting message bundle file with requested translations.
017     *
018     * @return {@link MessageBundle}. Message bundle with specified name should be exist.
019     */
020    MessageBundle getMessageBundle(String bundleName, Locale locale);
021}
