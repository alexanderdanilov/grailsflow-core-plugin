019package org.christianschenk.simplecache;
020
021import java.util.Collections;
022import java.util.HashMap;
023import java.util.Map;
024import java.util.concurrent.ExecutorService;
025import java.util.concurrent.Executors;
026import java.util.concurrent.TimeUnit;
027
028/**
029 * This class provides a very simple implementation of an object cache.
030 * 
031 * @author Christian Schenk
032 */
033public class SimpleCache<T> {
034
035        /** Objects are stored here */
036        private final Map<String, T> objects;
037        /** Holds custom expiration dates */
038        private final Map<String, Long> expire;
039        /** The default expiration date */
040        private final long defaultExpire;
041        /** Is used to speed up some operations */
042        private final ExecutorService threads;
043
044        /**
045         * Constructs the cache with a default expiration time for the objects of
046         * 100 seconds.
047         */
048        public SimpleCache() {
049                this(100);
050        }
051
052        /**
053         * Construct a cache with a custom expiration date for the objects.
054         * 
055         * @param defaultExpire
056         *            default expiration time in seconds
057         */
058        public SimpleCache(final long defaultExpire) {
059                this.objects = Collections.synchronizedMap(new HashMap<String, T>());
060                this.expire = Collections.synchronizedMap(new HashMap<String, Long>());
061
062                this.defaultExpire = defaultExpire;
063
064                this.threads = Executors.newFixedThreadPool(256);
065                Executors.newScheduledThreadPool(2).scheduleWithFixedDelay(this.removeExpired(), this.defaultExpire / 2, this.defaultExpire, TimeUnit.SECONDS);
066        }
067
068        /**
069         * This Runnable removes expired objects.
070         */
071        private final Runnable removeExpired() {
072                return new Runnable() {
073                        public void run() {
074                                for (final String name : expire.keySet()) {
075                                        if (System.currentTimeMillis() > expire.get(name)) {
076                                                threads.execute(createRemoveRunnable(name));
077                                        }
078                                }
079                        }
080                };
081        }
082
083        /**
084         * Returns a runnable that removes a specific object from the cache.
085         * 
086         * @param name
087         *            the name of the object
088         */
089        private final Runnable createRemoveRunnable(final String name) {
090                return new Runnable() {
091                        public void run() {
092                                objects.remove(name);
093                                expire.remove(name);
094                        }
095                };
096        }
097
098        /**
099         * Returns the default expiration time for the objects in the cache.
100         * 
101         * @return default expiration time in seconds
102         */
103        public long getExpire() {
104                return this.defaultExpire;
105        }
106
107        /**
108         * Put an object into the cache.
109         * 
110         * @param name
111         *            the object will be referenced with this name in the cache
112         * @param obj
113         *            the object
114         */
115        public void put(final String name, final T obj) {
116                this.put(name, obj, this.defaultExpire);
117        }
118
119        /**
120         * Put an object into the cache with a custom expiration date.
121         * 
122         * @param name
123         *            the object will be referenced with this name in the cache
124         * @param obj
125         *            the object
126         * @param expire
127         *            custom expiration time in seconds
128         */
129        public void put(final String name, final T obj, final long expireTime) {
130                this.objects.put(name, obj);
131                this.expire.put(name, System.currentTimeMillis() + expireTime * 1000);
132        }
133
134        /**
135         * Returns an object from the cache.
136         * 
137         * @param name
138         *            the name of the object you'd like to get
139         * @param type
140         *            the type of the object you'd like to get
141         * @return the object for the given name and type
142         */
143        public T get(final String name) {
144                final Long expireTime = this.expire.get(name);
145                if (expireTime == null) return null;
146                if (System.currentTimeMillis() > expireTime) {
147                        this.threads.execute(this.createRemoveRunnable(name));
148                        return null;
149                }
150                return this.objects.get(name);
151        }
152
153        /**
154         * Convenience method.
155         */
156        @SuppressWarnings("unchecked")
157        public <R extends T> R get(final String name, final Class<R> type) {
158                return (R) this.get(name);
159        }
160}
